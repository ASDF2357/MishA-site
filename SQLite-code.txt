CREATE TABLE "Users" (
	"USER_ID"	INTEGER NOT NULL,
	"LOGIN"	TEXT NOT NULL UNIQUE,
	"PASSWORD"	TEXT NOT NULL,
	PRIMARY KEY("USER_ID" AUTOINCREMENT)
);
CREATE TABLE "Adresses" (
	"ADRESS_ID"	INTEGER,
	"USER_ID"	INTEGER NOT NULL,
	"STREET"	INTEGER NOT NULL,
	"HOUSE"	INTEGER NOT NULL,
	"BUILDING"	TEXT NOT NULL,
	FOREIGN KEY("USER_ID") REFERENCES "Users"("USER_ID"),
	PRIMARY KEY("ADRESS_ID" AUTOINCREMENT)
);
CREATE TABLE "Orders" (
	"ORDER_ID"	INTEGER,
	"USER_ID"	INTEGER NOT NULL,
	"ADRESS_ID"	INTEGER NOT NULL,
	"DATA"	TEXT NOT NULL,
	"STATUS"	TEXT NOT NULL,
	PRIMARY KEY("ORDER_ID" AUTOINCREMENT),
	FOREIGN KEY("USER_ID") REFERENCES "Users"("USER_ID"),
	FOREIGN KEY("ADRESS_ID") REFERENCES "Adresses"("ADRESS_ID")
);

CREATE TABLE "Products" (
	"PRODUCTS_ID"	INTEGER,
	"CATEGORY"	TEXT NOT NULL,
	"ABOUT"	INTEGER,
	PRIMARY KEY("PRODUCTS_ID" AUTOINCREMENT)
);
CREATE TABLE "Prises" (
	"PRISE_ID"	INTEGER,
	"PRODUCT_ID"	INTEGER NOT NULL,
	"DATA"	TEXT NOT NULL,
	PRIMARY KEY("PRISE_ID" AUTOINCREMENT)
	FOREIGN KEY("PRODUCT_ID") REFERENCES "Products"("PRODUCT_ID")
);
CREATE TABLE "Order_list" (
	"ORDER_LIST_ID"	INTEGER,
	"ORDER_ID"	INTEGER NOT NULL,
	"PRODUCT_ID"	INTEGER NOT NULL,
	"COUNT"	INTEGER NOT NULL,
	"PRISE_ID"	INTEGER NOT NULL,
	PRIMARY KEY("ORDER_LIST_ID" AUTOINCREMENT)
	FOREIGN KEY("ORDER_ID") REFERENCES "Orders"("ORDER_ID"),
	FOREIGN KEY("PRODUCT_ID") REFERENCES "Products"("PRODUCT_ID")
);
